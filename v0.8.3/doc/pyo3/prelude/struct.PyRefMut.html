[−][src]Struct pyo3::prelude::PyRefMut

pub struct PyRefMut<'a, T: PyTypeInfo>(_, _);

Mutable version of PyRef.

Example

use pyo3::prelude::*;
use pyo3::types::IntoPyDict;
#[pyclass]
struct Point {
    x: i32,
    y: i32,
}
#[pymethods]
impl Point {
    fn length(&self) -> i32 {
        self.x * self.y
    }
}
let gil = Python::acquire_gil();
let py = gil.python();
let mut obj = PyRefMut::new(gil.python(), Point { x: 3, y: 4 }).unwrap();
let d = vec![("p", obj.to_object(py))].into_py_dict(py);
obj.x = 5; obj.y = 20;
py.run("assert p.length() == 100", None, Some(d)).unwrap();
